@startuml

entity User {
    id: INT
    name: VARCHAR
    current_location: INT
    calendar: INT
}

entity Calendar {
    id: INT
}

entity Event {
    id: INT
    name: VARCHAR
    location: INT
    start_time: TIMESTAMP
    end_time: TIMESTAMP
    duration: INTERVAL
}

entity Task {
    id: INT
    description: TEXT
    location: INT
    priority: BOOLEAN
    indoor_task: BOOLEAN
    emergency_task: BOOLEAN
    needs_display: BOOLEAN
}

entity Location {
    id: INT
    latitude: DOUBLE
    longitude: DOUBLE
}

entity Route {
    id: INT
    start: INT
    end: INT
    accessible: BOOLEAN
    path: TEXT
    type: VARCHAR
}

entity Room {
    id: INT
    number: VARCHAR
    type: VARCHAR
    accessibility_info: INT
}

entity Campus {
    id: INT
    name: VARCHAR
}

entity Building {
    id: INT
    name: VARCHAR
    campus: INT
}

entity Floor {
    id: INT
    number: INT
    building: INT
}

entity AccessibilityInfo {
    id: INT
    is_wheelchair_accessible: BOOLEAN
    has_elevator_access: BOOLEAN
    has_ramp_access: BOOLEAN
}

entity POI {
    id: INT
    type: VARCHAR
    location: INT
}

' Define Relationships
User }o--|| Calendar
User }o--|| Location
Event }o--|| Location
Task }o--|| Location
Building }o--|| Campus
Floor }o--|| Building
Room }o--|| AccessibilityInfo
POI }o--|| Location
Route }o--|| Location : start
Route }o--|| Location : end

@enduml
